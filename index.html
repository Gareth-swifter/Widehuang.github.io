####2.ImageIO.framework

利用ImageIO框架对读取到的gif进行处理，读取到每一帧的图片存入数组，然后交给UIImageView处理。

NSString* filePath = [[NSBundle mainBundle] pathForResource:@"1.gif" ofType:nil];
NSData* imageData = [NSData dataWithContentsOfFile:filePath];

CGImageSourceRef source = CGImageSourceCreateWithData((CFDataRef)imageData, NULL);
size_t count = CGImageSourceGetCount(source);

NSMutableArray* tmpArray = [NSMutableArray array];

for (size_t i = 0; i < count; i++) {

    CGImageRef cgimage = CGImageSourceCreateImageAtIndex(source, i, NULL);
    UIImage  *image = [UIImage imageWithCGImage:cgimage scale:[UIScreen mainScreen].scale orientation:UIImageOrientationUp];
    [tmpArray addObject:image];

    CGImageRelease(cgimage);
}
CFRelease(source);

UIImageView *gifImageView = [[UIImageView alloc] initWithFrame:[[UIScreen mainScreen] bounds]];
gifImageView.animationImages = tmpArray; //动画图片数组
gifImageView.animationDuration = tmpArray.count/24; //执行一次完整动画所需的时长
//gifImageView.animationRepeatCount = 6;  //动画重复次数
[gifImageView startAnimating];
[self.view addSubview:gifImageView];

这个全部都都使用Core Graphics的API，也可以使用UIGraphics和CG一起来创建.对上面的函数只需要替换一些函数。

UIImageView *imageView = [[UIImageView alloc] initWithFrame:CGRectMake(0, 0, 100, 100)];
imageView.center = CGPointMake(200, 300);
UIImage *anotherImage = [UIImage imageNamed:@"image.jpg"];
[self.view addSubview:imageView];

int w = imageView.frame.size.width;
int h = imageView.frame.size.height;

//替换了创建上下文的函数
UIGraphicsBeginImageContextWithOptions(imageView.bounds.size, NO, 1.0);
CGContextRef context = UIGraphicsGetCurrentContext();

CGRect rect = CGRectMake(0, 0, w, h);

CGContextSaveGState(context);
//平移变换
CGContextTranslateCTM(context, 0, 0);
//缩放
CGContextScaleCTM(context, 20, 20);
//绘制圆角路径
float fw, fh;
fw = CGRectGetWidth(rect) / 20;
fh = CGRectGetHeight(rect) / 20;
CGContextBeginPath(context);
CGContextMoveToPoint(context, fw, fh/2);  // Start at lower right corner
CGContextAddArcToPoint(context, fw, fh, fw/2, fh, 1);  // Top right corner
CGContextAddArcToPoint(context, 0, fh, 0, fh/2, 1); // Top left corner
CGContextAddArcToPoint(context, 0, 0, fw/2, 0, 1); // Lower left corner
CGContextAddArcToPoint(context, fw, 0, fw, fh/2, 1); // Back to lower right
CGContextClosePath(context);
CGContextRestoreGState(context);

//添加了平移和翻转的函数
CGContextTranslateCTM(context, 0, 100);
CGContextScaleCTM(context, 1, -1);

CGContextClip(context);
CGContextDrawImage(context, CGRectMake(0, 0, w, h), anotherImage.CGImage);
anotherImage = UIGraphicsGetImageFromCurrentImageContext();

CGContextRelease(context);

//结束绘制的函数
UIGraphicsEndImageContext();

imageView.image = anotherImage;

相对于纯CG的API绘制出的图片，多了一次的翻转（缩放的倍数为-1），如果不做这一步则生成图片是倒着的。这是由于UI框架里的坐标系与CG的是上下相反的，UI在左上角，而CG的在左下角。所以又做了一次翻转。

结果图：

最后还是发现 这个图模糊了。。

第一阶段总结：设计出4种将UIImageView变换圆角的方法（第三种和第四种貌似差不多），由于使用图片比绘制的图形大很多，所以个人觉得第三种和第四种在绘制图形时候，系统将图片先进行了缩放，然后再绘制，所以才会出现模糊的情况。有时间再测试一下相同大小（绘制大小与图片大小相同）时候会不会出现这样的问题。

方案二: 添加动画的监听

//UIViewController+Animating.h
/**
*  监听push是否在执行.
*  */
@property (nonatomic,assign) BOOL animating;

//UIViewController+Animating.m
/**
*    set和get函数实现
*  */
static const void * Animating = &Animating;

- (void)setAnimating:(BOOL)animating{
    objc_setAssociatedObject(self, &Animating, @(animating), OBJC_ASSOCIATION_ASSIGN);
    }
- (BOOL)animating{
     NSNumber * animating = objc_getAssociatedObject(self, &Animating);
    return  [animating boolValue];
}

-

//在界面出现和消失时候将animating设置为No
[UIViewController swizzleSelector:@selector(viewDidAppear:) originalSelector:@selector(rt_viewDidAppear:)];
[UIViewController swizzleSelector:@selector(viewDidDisappear:) originalSelector:@selector(rt_viewDidDisappear:)];
- (void)rt_viewDidAppear:(BOOL)animated {
    [self rt_viewDidAppear:animated];
    self.animating = NO;
}    
- (void)rt_viewDidDisappear:(BOOL)animated {
    [self rt_viewDidDisappear:animated];
    self.animating = NO;
}

-
//UINavigationController
/**

 *  重写了pushViewController的方法
 *  解决了push时候 因为push动画而导致的界面错误
 */
- (void)pushViewController:(UIViewController *)viewController animated:(BOOL)animated{
    if (animated) {
       if (self.topViewController.animating) {
               NSLog(@"%@ controller(%@),push when animating",NSStringFromSelector(_cmd),NSStringFromClass([self.topViewController class]));
            return;
        }
        self.topViewController.animating = YES;
        viewController.animating = YES;
    }
    [super pushViewController:viewController animated:animated];
}

个人觉得这样的实现应该是最OK的.如果有更好的方法可以联系我,我们一起探讨.
